```javascript
// Pseudo-code for scroll sync
const leftPane = document.querySelector('.comparison-pane.left .comparison-content');
const rightPane = document.querySelector('.comparison-pane.right .comparison-content');
let isSynced = true;

leftPane.addEventListener('scroll', () => {
  if (isSynced) {
    const scrollPercent = leftPane.scrollTop / (leftPane.scrollHeight - leftPane.clientHeight);
    rightPane.scrollTop = scrollPercent * (rightPane.scrollHeight - rightPane.clientHeight);
  }
});

rightPane.addEventListener('scroll', () => {
  if (isSynced) {
    const scrollPercent = rightPane.scrollTop / (rightPane.scrollHeight - rightPane.clientHeight);
    leftPane.scrollTop = scrollPercent * (leftPane.scrollHeight - leftPane.clientHeight);
  }
});

// Toggle sync
document.querySelector('[data-toggle-sync]').addEventListener('click', () => {
  isSynced = !isSynced;
});
```
